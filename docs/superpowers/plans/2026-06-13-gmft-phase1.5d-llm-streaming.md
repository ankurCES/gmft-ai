## Task 1: `createModel({provider, model, apiKey, endpoint?})`

**Files:**
- Create: `packages/core/src/llm/model-factory.ts`

The factory returns a Vercel AI SDK `LanguageModelV1` (or whatever the SDK exposes as the "model handle" — the type is `LanguageModel` in v4). One branch per provider id; `openai-compatible` handles both `openrouter` and `ollama`.

```ts
import { createAnthropic } from '@ai-sdk/anthropic';
import { createOpenAI } from '@ai-sdk/openai';
import { createGoogleGenerativeAI } from '@ai-sdk/google';
import { createOpenAICompatible } from '@ai-sdk/openai-compatible';
import type { LanguageModel } from 'ai';
import type { LlmConfig } from '../config/config.js';

export interface CreateModelOpts {
  provider: LlmConfig['provider'];
  model: string;
  apiKey: string;            // empty string for endpoint-only providers (ollama)
  endpoint?: string;         // required for openrouter, ollama
}

export function createModel(opts: CreateModelOpts): LanguageModel {
  const { provider, model, apiKey, endpoint } = opts;
  switch (provider) {
    case 'anthropic': {
      const a = createAnthropic({ apiKey });
      return a(model);
    }
    case 'openai': {
      const o = createOpenAI({ apiKey });
      return o(model);
    }
    case 'google': {
      const g = createGoogleGenerativeAI({ apiKey });
      return g(model);
    }
    case 'openrouter': {
      if (!endpoint) throw new Error('openrouter requires endpoint');
      const c = createOpenAICompatible({
        name: 'openrouter',
        apiKey,
        baseURL: endpoint,
      });
      return c(model);
    }
    case 'ollama': {
      if (!endpoint) throw new Error('ollama requires endpoint');
      const c = createOpenAICompatible({
        name: 'ollama',
        apiKey: apiKey || 'ollama', // SDK requires a non-empty key
        baseURL: endpoint,
      });
      return c(model);
    }
  }
}
```

## Task 3: `runTurn({model, history, system, signal})` — the agent loop (no tools, v0.1)

**Files:**
- Create: `packages/core/src/agent/loop.ts`
- Create: `packages/core/test/agent-loop.test.ts`

v0.1 has no tools (chokepoint lands in phase 3). `runTurn` is a thin wrapper around Vercel AI SDK's `streamText` that:
1. Builds the `messages` array from `history: ChatMessage[]`.
2. Calls `streamText({ model, system, messages, abortSignal: signal, maxSteps: 1 })` (no tool calls possible → one step is enough).
3. Returns an `AsyncIterable<AgentEvent>` of `text-delta` chunks, then a `done` event.

```ts
import { streamText, type LanguageModel, type CoreMessage } from 'ai';
import type { ChatMessage } from './context.js';

export type AgentEvent =
  | { type: 'text-delta'; text: string }
  | { type: 'done'; finishReason: string; text: string }
  | { type: 'error'; error: Error };

export interface RunTurnOpts {
  model: LanguageModel;
  system: string;
  history: readonly ChatMessage[];
  signal?: AbortSignal;
}

export async function* runTurn(opts: RunTurnOpts): AsyncIterable<AgentEvent> {
  const messages: CoreMessage[] = opts.history.map((m) => ({
    role: m.role,
    content: m.content,
  }));
  let fullText = '';
  try {
    const result = streamText({
      model: opts.model,
      system: opts.system,
      messages,
      abortSignal: opts.signal,
      maxSteps: 1,
    });
    for await (const chunk of result.fullStream) {
      if (chunk.type === 'text-delta') {
        fullText += chunk.textDelta;
        yield { type: 'text-delta', text: chunk.textDelta };
      }
      // ignore other chunk types for v0.1
    }
    yield { type: 'done', finishReason: 'stop', text: fullText };
  } catch (err) {
    yield { type: 'error', error: err instanceof Error ? err : new Error(String(err)) };
  }
}
```

`index.ts` re-export: `export { runTurn, type AgentEvent, type RunTurnOpts } from './agent/loop.js';`

```ts
export type ChatRole = 'system' | 'user' | 'assistant';

export interface ChatMessage {
  role: ChatRole;
  content: string;
  ts?: number;
}

export function tokenEstimate(text: string): number {
  // v0.1 approximation: 1 token ≈ 4 chars of English text. Tighter models
  // (cl100k_base tiktoken) are ~1 token ≈ 3.7 chars; ±15% is fine for the
  // v0.1 summarizer budget guard. Phase 3 swaps this for tiktoken.
  return Math.ceil(text.length / 4);
}

export function totalTokens(messages: readonly ChatMessage[]): number {
  let n = 0;
  for (const m of messages) n += tokenEstimate(m.content) + 4; // +4 for role/format
  return n;
}
```

## Task 5: `summarizeIfNeeded(history, budget)`

**Files:**
- Create: `packages/core/src/agent/summarizer.ts`
- Create: `packages/core/test/summarizer.test.ts`

```ts
import type { ChatMessage } from './context.js';
import { totalTokens, tokenEstimate } from './context.js';

export interface SummarizeResult {
  /** Updated history (truncated from the front + possibly a synthetic summary). */
  history: ChatMessage[];
  /** True iff a summary was generated. v0.1 stub returns a static marker. */
  summarized: boolean;
  /** Tokens of the new history. */
  tokens: number;
}

export interface SummarizeOpts {
  history: readonly ChatMessage[];
  /** Max total tokens allowed in the returned history. */
  budget: number;
  /** v0.1: stub. Phase 2 wires a real LLM call here. */
  generateSummary?: (chunk: readonly ChatMessage[]) => Promise<string>;
}

/**
 * If `history` fits under `budget` tokens, return it as-is. Otherwise,
 * drop the oldest messages until the budget is met. If a `generateSummary`
 * callback is provided, prepend a synthetic "system" message summarizing
 * the dropped chunk; otherwise just drop them.
 */
export async function summarizeIfNeeded(opts: SummarizeOpts): Promise<SummarizeResult> {
  const { history, budget, generateSummary } = opts;
  if (totalTokens(history) <= budget) {
    return { history: [...history], summarized: false, tokens: totalTokens(history) };
  }
  // Drop from the front until under budget. Always keep at least the last
  // message (don't return an empty list).
  const dropped: ChatMessage[] = [];
  let kept = [...history];
  while (kept.length > 1 && totalTokens(kept) > budget) {
    dropped.push(kept.shift()!);
  }
  let summary: string | null = null;
  if (generateSummary && dropped.length > 0) {
    summary = await generateSummary(dropped);
  }
  const final: ChatMessage[] = summary
    ? [{ role: 'system', content: `[Earlier summary] ${summary}` }, ...kept]
    : kept;
  return { history: final, summarized: dropped.length > 0, tokens: totalTokens(final) };
}
```

`index.ts` re-export: `export { summarizeIfNeeded, type SummarizeOpts, type SummarizeResult } from './agent/summarizer.js';`

**Test** (`summarizer.test.ts`): 1 file, 1 test (covers 2 scenarios).

- **No-op case:** `summarizeIfNeeded({history: [{role:'user', content:'hi'}], budget: 1000})` → `{summarized: false, tokens: ~5, history.length: 1}`.
- **Truncation case:** Build 50 messages of `'a'.repeat(100)` each (≈25 tokens each → 1250 total). `summarizeIfNeeded({history, budget: 200})` → `summarized: true`, returned history has way fewer messages, `tokens < 200`.
- **Summary callback case:** Same 50 messages, but pass `generateSummary: async (chunk) => 'prior context'` → returned history has a leading `{role:'system', content: '[Earlier summary] prior context'}` message, plus the trailing kept messages.

## Task 6: `useAgent` hook in `apps/gmft`

**Files:**

A small React hook that owns the LLM streaming state and returns a `(value: string) => void` submit function. Lives in `apps/gmft` (not `@gmft/core`) because it depends on React state and the test renderer.

```tsx
// apps/gmft/src/ui/hooks/useAgent.ts
import { useCallback, useRef, useState } from 'react';
import { runTurn, buildSystemPrompt, type AgentEvent } from '@gmft/core';
import type { CreateModelOpts } from '@gmft/core';
import type { ChatMessage, PromptEnv } from '@gmft/core';

export interface UseAgentOpts {
  model: CreateModelOpts;
  env: PromptEnv;
  initialHistory?: readonly ChatMessage[];
  onError?: (err: Error) => void;
}

export interface UseAgentResult {
  history: readonly ChatMessage[];
  streaming: boolean;
  error: Error | null;
  submit: (text: string) => void;
  abort: () => void;
}

export function useAgent(opts: UseAgentOpts): UseAgentResult {
  const [history, setHistory] = useState<ChatMessage[]>(
    [...(opts.initialHistory ?? [])],
  );
  const [streaming, setStreaming] = useState(false);
  const [error, setError] = useState<Error | null>(null);
  const abortRef = useRef<AbortController | null>(null);

  const submit = useCallback(
    (text: string) => {
      const trimmed = text.trim();
      if (!trimmed || streaming) return;
      setError(null);
      setStreaming(true);
      const userMsg: ChatMessage = { role: 'user', content: trimmed, ts: Date.now() };
      const assistantMsg: ChatMessage = { role: 'assistant', content: '', ts: Date.now() };
      setHistory((h) => [...h, userMsg, assistantMsg]);
      // Dynamic import: keep `@gmft/core`'s LLM SDK deps out of the test
      // bundle when the test uses a fake model injected through `runTurn`.
      // (For v0.1, we statically import createModel + runTurn because the
      // SDK packages are small and tree-shakable. Switch to dynamic import
      // only if a Vitest test fails on a missing native dep.)
      void (async () => {
        const { createModel } = await import('@gmft/core');
        const model = createModel(opts.model);
        const system = buildSystemPrompt('agent', opts.env);
        const ac = new AbortController();
        abortRef.current = ac;
        try {
          for await (const ev of runTurn({
            model,
            system,
            history: [...history, userMsg],
            signal: ac.signal,
          })) {
            if (ac.signal.aborted) break;
            if (ev.type === 'text-delta') {
              setHistory((h) => {
                const next = [...h];
                const last = next[next.length - 1];
                if (last && last.role === 'assistant') {
                  next[next.length - 1] = { ...last, content: last.content + ev.text };
                }
                return next;
              });
            } else if (ev.type === 'error') {
              setError(ev.error);
            }
          }
        } finally {
          setStreaming(false);
          abortRef.current = null;
        }
      })();
    },
    [history, streaming, opts.model, opts.env],
  );

  const abort = useCallback(() => {
    abortRef.current?.abort();
  }, []);

  return { history, streaming, error, submit, abort };
}
```

> **Dependency-injection seam for tests:** the test should NOT exercise real `createModel` (it would load the AI SDK providers, which might require API keys at import time). Instead, the test mocks `@gmft/core`'s `runTurn` via `vi.mock('@gmft/core', ...)` to return a fake `AsyncIterable`. The `useAgent` body calls `runTurn` (not `createModel` directly), so the test can replace the whole `runTurn` with a deterministic generator. `createModel` is still imported in the real hook for production; the test's `vi.mock` stub doesn't need to provide it because `useAgent` references it via the dynamic `import('@gmft/core')`.

```tsx
import { useMemo } from 'react';
import { createModel, buildSystemPrompt, runTurn, type ChatMessage, type CreateModelOpts, type PromptEnv } from '@gmft/core';
import { App, type AppProps } from './App.js';
import type { Message as Msg } from './ui/components/Message.js';

export interface AgentAppProps extends Omit<AppProps, 'onSubmit'> {
  model: CreateModelOpts;
  env: PromptEnv;
  /** Hook to receive streamed chunks for tests; production passes nothing. */
  onDelta?: (delta: string) => void;
}

export function AgentApp({ model, env, onDelta, ...appProps }: AgentAppProps): React.JSX.Element {
  const system = useMemo(() => buildSystemPrompt('agent', env), [env]);
  const llmModel = useMemo(() => createModel(model), [model]);
  const handleSubmit = async (value: string): Promise<Msg | null> => {
    let buffer = '';
    try {
      for await (const ev of runTurn({
        model: llmModel,
        system,
        history: appProps.initialHistory?.map((s) => ({ role: 'user', content: s })) ?? [],
      })) {
        if (ev.type === 'text-delta') {
          buffer += ev.text;
          onDelta?.(ev.text);
        } else if (ev.type === 'error') {
          return {
            id: `err-${Date.now()}`,
            role: 'assistant',
            content: `[error] ${ev.error.message}`,
            ts: Date.now(),
          };
        }
      }
      return {
        id: `a-${Date.now()}`,
        role: 'assistant',
        content: buffer || '(empty response)',
        ts: Date.now(),
      };
    } catch (err) {
      return {
        id: `err-${Date.now()}`,
        role: 'assistant',
        content: `[error] ${err instanceof Error ? err.message : String(err)}`,
        ts: Date.now(),
      };
    }
  };
  return <App {...appProps} onSubmit={handleSubmit} />;
}
```

## Risks

1. **Vercel AI SDK v4 API churn:** the `streamText` signature, the `fullStream` chunk types, and the `LanguageModelV1` interface may have moved between 4.0 and 4.3.19. The plan body above uses 4.3.19's actual signatures. **If a future SDK upgrade changes any of these, the tests catch it immediately because they're integration-style (real SDK calls with a fake model).**

2. **`@ai-sdk/openai-compatible` baseURL semantics:** the SDK's `baseURL` for OpenRouter/Ollama expects the path-without-`/v1` in some versions and `with-/-v1` in others.
