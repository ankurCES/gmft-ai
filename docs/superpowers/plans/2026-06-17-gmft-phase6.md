# GMFT-AI Phase 6 Implementation Plan

**Date:** 2026-06-17
**Spec:** `docs/superpowers/specs/2026-06-17-gmft-phase6-design.md` (584 lines, commit 5c7e166)
**Test budget:** 288 → 319 (+31 tests; 6/7/13/5 per feature)
**Branch:** `main` (no sub-branch — small enough to land in one PR)
**Order:** C (more wifi) → B (report + Findings) → A (attack chain) → D (scope file)

## How this plan is organized

Each feature gets a top-level section. Tasks within a feature are numbered
(`C.1`, `B.1`, …) and written so a single agent session can execute a
contiguous block in one shot. Each feature is bracketed by:
- a "depends on" line (the previous feature's last commit, or the spec),
- a "delivers" line (cumulative test count + commit count).

Test budget is enforced by a `test-count-guard` task in §11 that runs
after every feature and aborts the session if the count drifts.

## Conventions

- All file paths in tasks are repo-relative (`packages/core/src/...`).
- `pnpm vitest run` is the test command unless the task says otherwise.
- Type-check after each feature: `pnpm -r typecheck`.
- Each feature ends with a commit; commit messages are provided verbatim.
- TDD: write the failing test first, then the impl, then refactor.
- The "decision deferred" items in spec §9 are resolved in the relevant
  feature's first task (not as standalone tasks — they're "decide when
  writing the per-feature task").

## Open-question resolutions (from spec §9)

| # | Question | Resolution | Lands in |
|---|----------|------------|----------|
| 1 | chain-step audit log format (JSONL one-line-per-step or one entry per chain?) | One entry per chain, with a nested `steps: [{tool, args, status, findingCount, durationMs}]` array. The chain `chainId` is the parent audit id; each step's `findingId`s are cross-referenced. JSONL stays one-line-per-finding (existing contract). | A.1 |
| 2 | `report_write` output path policy | Default: `~/.local/share/gmft/reports/{sessionId}.{ext}`. Tool arg `outputPath?: string` overrides. Path is normalized + verified to be inside the reports dir (no `..` escape) before write. | B.1 |
| 3 | `targetsFromFile` argv substitution | Executor clones `args` per target and **replaces** the `target` field with the line value. Other args stay constant. Simple, predictable, matches the brainstorming decision. | D.1 |

## A.2 — `attack_chain` tool

**File:** `packages/tools/src/chains/attack-chain.ts` (new, ~180 lines)

```ts
export const ATTACK_CHAIN_TOOL: Tool<...> = {
  name: 'attack_chain',
  description: 'Execute a sequence of tools as a single confirmed operation. ...',
  inputSchema: z.object({
    steps: z.array(z.object({
      tool: z.string(),
      args: z.record(z.unknown()),
      name: z.string().optional(),
    })).min(1).max(20),                          // cap at 20 steps to mitigate the "LLM tricked into 20-step chain" risk
    stopOnDeny: z.boolean().default(true),
  }),
  outputSchema: z.object({
    chainId: z.string(),
    steps: z.array(z.object({
      tool: z.string(),
      status: z.enum(['ok', 'denied', 'erred', 'skipped']),
      findingCount: z.number(),
      durationMs: z.number(),
      reason: z.string().optional(),
    })),
    totals: z.object({
      total: z.number(),
      completed: z.number(),
      denied: z.number(),
      erred: z.number(),
      skipped: z.number(),
    }),
  }),
  flags: {
    destructive: true,
    requiresElevation: true,
    typeToConfirm: 'attack',                    // covers the whole chain
  },
  async run(args, ctx) {
    const chainId = crypto.randomUUID();
    const results = [];
    let skipped = 0, denied = 0, erred = 0, completed = 0;

    ctx.emit({ type: 'chain-started', chainId, stepCount: args.steps.length });

    for (let i = 0; i < args.steps.length; i++) {
      const step = args.steps[i];
      ctx.emit({ type: 'chain-step-started', chainId, stepIndex: i, tool: step.tool, name: step.name });

      const start = Date.now();
      let result: ExecuteResult;
      try {
        result = await ctx.innerRunner(step.tool, step.args, { suppressTypeToConfirm: true });
      } catch (err) {
        const durationMs = Date.now() - start;
        ctx.emit({ type: 'chain-step-finished', chainId, stepIndex: i, status: 'erred', durationMs, findingCount: 0, reason: String(err) });
        results.push({ tool: step.tool, status: 'erred', findingCount: 0, durationMs, reason: String(err) });
        erred++;
        if (args.stopOnDeny) break;
        continue;
      }

      const durationMs = Date.now() - start;
      const status = result.denied ? 'denied' : (result.error ? 'erred' : 'ok');
      // (The 'skipped' status happens when stopOnDeny && previous step was denied/erred.)
      if (status === 'denied') denied++;
      else if (status === 'erred') erred++;
      else completed++;

      ctx.emit({ type: 'chain-step-finished', chainId, stepIndex: i, status, durationMs, findingCount: result.findings?.length ?? 0, reason: result.denied ?? result.error });

      results.push({ tool: step.tool, status, findingCount: result.findings?.length ?? 0, durationMs, reason: result.denied ?? result.error });

      if (args.stopOnDeny && status !== 'ok') {
        // mark remaining steps as skipped
        for (let j = i + 1; j < args.steps.length; j++) {
          results.push({ tool: args.steps[j].tool, status: 'skipped', findingCount: 0, durationMs: 0 });
          ctx.emit({ type: 'chain-step-finished', chainId, stepIndex: j, status: 'skipped', durationMs: 0, findingCount: 0, reason: 'previous step failed' });
          skipped++;
        }
        break;
      }
    }

    ctx.emit({ type: 'chain-finished', chainId, totalSteps: args.steps.length, completed, denied, erred });

    // Audit log: write a single entry with nested steps (per spec §9 #1)
    await ctx.audit?.log({
      action: 'attack_chain',
      chainId,
      steps: results,
      totals: { total: args.steps.length, completed, denied, erred, skipped },
    });

    return { chainId, steps: results, totals: { total: args.steps.length, completed, denied, erred, skipped } };
  },
};
```

**Open-question #1 resolution (lands here):** Audit log uses one entry
per chain with nested `steps: [...]`. The session's `findings.jsonl`
still gets one line per finding (existing contract preserved). The
chain's `chainId` is the audit entry's id; each step's findingIds
are cross-referenced in the audit entry's `steps[].findingIds`.

2. **mid-chain deny (step 2 denied, step 3 skipped)** — with

## D.1 — `readTargetsFile` helper

**File:** `packages/core/src/tools/targets-file.ts` (new, ~50 lines)

```ts
/**
 * Read a targets file: one target per line, `#` comments, blank lines
 * ignored. Throws on:
 *   - missing file
 *   - more than 10k lines (v0.1 cap)
 *   - file size > 1MB
 */
export async function readTargetsFile(path: string): Promise<string[]> {
  const stat = await fs.stat(path);
  if (stat.size > 1024 * 1024) throw new Error(`targets file too large: ${stat.size} bytes`);
  const content = await fs.readFile(path, 'utf-8');
  const lines = content.split('\n')
    .map(l => l.trim())
    .filter(l => l.length > 0 && !l.startsWith('#'));
  if (lines.length > 10_000) throw new Error(`targets file has too many lines: ${lines.length} (max 10000)`);
  return lines;
}
```

# §11 — test-count-guard

A small script to catch test-count drift early. Lives at
`scripts/test-count-guard.mjs` (new, ~30 lines).

```js
#!/usr/bin/env node
import { execSync } from 'node:child_process';
const out = execSync('pnpm vitest run --reporter=json', { encoding: 'utf-8' });
// vitest JSON reporter outputs JSON per-file; aggregate numTotalTests
const lines = out.split('\n').filter(l => l.startsWith('{'));
const total = lines.reduce((acc, l) => {
  try { return acc + (JSON.parse(l).numTotalTests ?? 0); } catch { return acc; }
}, 0);
console.log(`Test count: ${total}`);
// compare against expected (read from an env var set by CI, or hardcode the
// expected count for the current feature)
const expected = Number(process.env.EXPECTED_TESTS ?? 0);
if (expected && total !== expected) {
  console.error(`Drift: expected ${expected}, got ${total}`);
  process.exit(1);
}
```

**Usage during execution:**

```bash
# After Feature C:

# After Feature B:
EXPECTED_TESTS=301 pnpm node scripts/test-count-guard.mjs

# After Feature A:
EXPECTED_TESTS=314 pnpm node scripts/test-count-guard.mjs

# After Feature D:
EXPECTED_TESTS=319 pnpm node scripts/test-count-guard.mjs
```

This is a development-time guard, not a CI check. CI uses `pnpm vitest run`
and asserts pass/fail only.
